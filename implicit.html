DECLARE
    -- Define a record type to match the structure of the emp table
    TYPE emp_record_type IS RECORD (
        empno INT,
        emp_name VARCHAR(50)
    );

    -- Declare a variable of the record type
    emp_rec emp_record_type;
BEGIN
    -- Open the implicit cursor to fetch employee IDs and names
    FOR emp_rec IN (SELECT empno, emp_name FROM emp) LOOP
        -- Display employee ID and name
        DBMS_OUTPUT.PUT_LINE('Employee ID: ' || emp_rec.empno || ', Employee Name: ' || emp_rec.emp_name);
    END LOOP;

    -- Display the total number of rows fetched
    DBMS_OUTPUT.PUT_LINE('Total number of employees: ' || SQL%ROWCOUNT);
END;
/
