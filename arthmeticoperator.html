ARITHMETIC OPERATORs

SUBTRACTIONS 

mysql> select 100-40;
+--------+
| 100-40 |
+--------+
|     60 |
+--------+
1 row in set (0.09 sec)


ADDITION 

mysql> select 100+10;
+--------+
| 100+10 |
+--------+
|    110 |
+--------+
1 row in set (0.07 sec)

MULTIPLE 


mysql> select 100*10;
+--------+
| 100*10 |
+--------+
|   1000 |
+--------+
1 row in set (0.00 sec)

DIVIDE 


mysql> select 100/10;
+---------+
| 100/10  |
+---------+
| 10.0000 |
+---------+
1 row in set (0.00 sec)

REMAINDER
 

mysql> select 100%10;
+--------+
| 100%10 |
+--------+
|      0 |
+--------+
1 row in set (0.00 sec)
