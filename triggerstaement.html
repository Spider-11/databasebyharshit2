 code

CREATE OR REPLACE TRIGGER statement_level_trigger
AFTER INSERT ON emp
DECLARE
    v_total_salary DECIMAL(10, 2);
BEGIN
    -- Calculate the total salary of all employees after each insert
    SELECT SUM(sal) INTO v_total_salary FROM emp;
    
    -- Display the total salary
    DBMS_OUTPUT.PUT_LINE('Total salary of all employees: ' || v_total_salary);
END;
/

INSERT INTO emp (empno, emp_name, job, sal) VALUES (1, 'John Doe', 'Manager', 5000);




Output

Total salary of all employees: 49500

1 row(s) inserted.

0.06 seconds
