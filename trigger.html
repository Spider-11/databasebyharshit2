CREATE OR REPLACE TRIGGER check_salary_trigger
BEFORE INSERT ON emp
FOR EACH ROW
DECLARE

    salary_too_high_exception EXCEPTION;
BEGIN
  
    IF :NEW.sal > 10000 THEN
      
        RAISE salary_too_high_exception;
    END IF;
EXCEPTION
  
    WHEN salary_too_high_exception THEN
   
        DBMS_OUTPUT.PUT_LINE('Error: Salary exceeds the threshold.');
END;
/
