CREATE OR REPLACE TRIGGER after_insert_trigger
AFTER INSERT ON emplo
FOR EACH ROW
BEGIN
    DBMS_OUTPUT.PUT_LINE('New row inserted. Employee number: ' || :NEW.empno || ', Employee name: ' || :NEW.emp_name);
END;
/
INSERT INTO emplo (empno, emp_name, sal) VALUES (101, 'John Doe', 5000);

Output

New employee number: 101
New employee name: John Doe
New salary: 5000
New row inserted. Employee number: 101, Employee name: John Doe

1 row(s) inserted.

0.04 seconds
