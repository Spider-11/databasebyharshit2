CREATE OR REPLACE TRIGGER before_update_trigger
BEFORE UPDATE ON emplo
FOR EACH ROW
BEGIN
    IF UPDATING('sal') THEN
        RAISE_APPLICATION_ERROR(-20001, 'Updates to salary are not allowed.');
    END IF;
END;
/

UPDATE emplo SET sal = sal * 1.1;
