CREATE OR REPLACE TRIGGER after_update_trigger
AFTER UPDATE ON emplo
FOR EACH ROW
BEGIN
    DBMS_OUTPUT.PUT_LINE('Row updated. Employee number: ' || :NEW.empno || ', New salary: ' || :NEW.sal);
END;
/

UPDATE emplo SET emp_name = 'John Doe' WHERE empno = 101;

Output

Row updated. Employee number: 101, New salary: 5000

1 row(s) updated.

0.01 seconds
