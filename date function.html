DATE FUNCTION

SYSTEM DATE 

mysql> select now() from dual;

mysql> select now() from dual;

| now()               |
+---------------------+
| 2024-04-10 19:44:32 |
+---------------------+
1 row in set (0.00 sec)

ADD MONTH

mysql> SELECT NOW(), DATE_ADD(NOW(), INTERVAL 4 MONTH);
+---------------------+-----------------------------------+
| NOW()               | DATE_ADD(NOW(), INTERVAL 4 MONTH) |
+---------------------+-----------------------------------+
| 2024-04-10 20:13:57 | 2024-08-10 20:13:57               |
+---------------------+-----------------------------------+
1 row in set (0.08 sec)

MONTH BETWEEN

mysql> SELECT TIMESTAMPDIFF(MONTH, '1978-12-25', '1981-12-25') AS date1;
+-------+
| date1 |
+-------+
|    36 |
+-------+
1 row in set (0.00 sec)

NEXT DATE

mysql> SELECT DATE_ADD(NOW(), INTERVAL 1 DAY) AS next_day;
+---------------------+
| next_day            |
+---------------------+
| 2024-04-11 20:20:48 |
+---------------------+
1 row in set (0.00 sec)
