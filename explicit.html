EXPLICIT CURSOR

 code

DECLARE
    -- Declare variables to store employee details
    empno emp.empno%TYPE;
    emp_name emp.emp_name%TYPE;
    job emp.job%TYPE;
    mgr emp.mgr%TYPE;
    hiredate emp.hiredate%TYPE;
    sal emp.sal%TYPE;
    commission emp.commission%TYPE;
    deptno emp.deptno%TYPE;

    -- Declare the explicit cursor
    CURSOR emp_cursor IS
        SELECT empno, emp_name, job, mgr, hiredate, sal, commission, deptno
        FROM emp;
BEGIN
    -- Open the cursor
    OPEN emp_cursor;
    
    -- Fetch data from the cursor and process each row
    LOOP
        FETCH emp_cursor INTO empno, emp_name, job, mgr, hiredate, sal, commission, deptno;
        EXIT WHEN emp_cursor%NOTFOUND; -- Exit the loop when no more rows are found
        
        -- Display employee details
        DBMS_OUTPUT.PUT_LINE('Employee No: ' || empno || ', Name: ' || emp_name || ', Job: ' || job || ', Manager: ' || mgr || ', Hire Date: ' || hiredate || ', Salary: ' || sal || ', Commission: ' || commission || ', Dept No: ' || deptno);
    END LOOP;
    
    -- Close the cursor
    CLOSE emp_cursor;
END;
