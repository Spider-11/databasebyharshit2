BEFORE DELETE TRIGGER

 code

CREATE OR REPLACE TRIGGER before_delete_trigger
BEFORE DELETE ON emplo
FOR EACH ROW
BEGIN
    
    IF :OLD.empno = 101 THEN
        DBMS_OUTPUT.PUT_LINE('Deletion of employee with empno 101 is not allowed.');
     
    END IF;
END;
/
DELETE FROM emplo WHERE empno = 101;

Output

Deletion of employee with empno 101 is not allowed.

1 row(s) deleted.
