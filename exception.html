DECLARE
    v_emp_name emp.emp_name%TYPE;
BEGIN
    SELECT emp_name INTO v_emp_name FROM emp WHERE empno = 1000;
    DBMS_OUTPUT.PUT_LINE('Employee Name: ' || v_emp_name);
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('No employee found for empno 1000.');
END;
/
