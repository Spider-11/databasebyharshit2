SQL> SET SERVEROUTPUT ON;
CREATE OR REPLACE TRIGGER before_insert_trigger
BEFORE INSERT ON emplo
FOR EACH ROW
BEGIN
    -- Accessing the new values being inserted
    DBMS_OUTPUT.PUT_LINE('New employee number: ' || :NEW.empno);
    DBMS_OUTPUT.PUT_LINE('New employee name: ' || :NEW.emp_name);
    DBMS_OUTPUT.PUT_LINE('New salary: ' || :NEW.sal);

    -- Example: Setting a default value if the salary is not provided
    IF :NEW.sal IS NULL THEN
        :NEW.sal := 0;
    END IF;
END;
/

INSERT INTO emplo (empno, emp_name, sal) VALUES (101, 'Alice', 5000);


Output

New employee number: 101
New employee name: Alice
New salary: 5000

1 row(s) inserted.

0.03 seconds
